#include <stdio.h>
#include <math.h>

int main() {
    float x, y, resultado;

    // Entrada de dados
    printf("Informe o valor de X: ");
    scanf("%f", &x);

    printf("Informe o valor de Y: ");
    scanf("%f", &y);

    // Cálculo da potência
    resultado = pow(x, y);

    // Saída
    printf("Resultado: %.1f\n", resultado);

    return 0;
}
